incident_king_in_prussia = {
    event = king_in_prussia.2
	default_option = 1

	can_stop = {
		OR = {
			NOT = {
				exists = PRU
			}

				is_subject = yes
			}
			PRU = {
				government_rank = 2
			}
		}
	}

	0 = { # grant

		modifier = {
			factor = 100
			OR = {
				ai_attitude = {
					who = PRU

				}
				ai_attitude = {
					who = PRU

				}
				ai_attitude = {
					who = PRU
					attitude = attitude_allied
				}
			}
		}
	}

	1 = { # compromise
		factor = 1
		modifier = {
			factor = 100
			ai_attitude = {
				who = PRU
				attitude = attitude_neutral
			}
		}
	}

	1 = { # deny
		factor = 1
		modifier = {
			factor = 100
			OR = {
				ai_attitude = {
					who = PRU
					attitude = attitude_rivalry
				}
				ai_attitude = {
					who = PRU
					attitude = attitude_hostile
				}
				ai_attitude = {
					who = PRU
					attitude = attitude_threatened
				}
				ai_attitude = {
					who = PRU
					attitude = attitude_outraged
				}
				ai_attitude = {
					who = PRU
					attitude = attitude_disloyal
				}
				ai_attitude = {
					who = PRU
					attitude = attitude_rebellious
				}
			}
		}
	}
}

incident_pu_subject_joins_empire = {
	event = incident_generic.2
	default_option = 0

	can_stop = {
		OR = {
			NOT = {
				any_country = {
					has_country_flag = current_pu_incident_target_flag
				}
			}
			any_country = {
				has_country_flag = current_pu_incident_target_flag
				OR = {
					is_subject = no
					overlord = {
						is_part_of_hre = no
					}
				}
			}
		}
	}

	0 = { # join
		factor = 1
		modifier = {
			factor = 1000
			any_subject_country = {
				has_country_flag = current_pu_incident_target_flag
			}
		}
	}

	1 = { # denied
		factor = 1
	}
}
