country_event = {
  id = flavor_syr.3
  title = flavor_syr.3.t
  desc = flavor_syr.3.d
  picture = BIG_BOOK_eventPicture

  fire_only_once = yes

  trigger = {
    OR = {
      tag = SYR
      was_tag = SYR
    }
    OR = {
      has_country_flag = azeri_help
      has_country_flag = kurd_help
    }
    OR = {
      SRV = {
        is_subject_of = ROOT
      }
      ARD = {
        is_subject_of = ROOT
      }
    }
  }

  mean_time_to_happen = {
    months = 120
  }

  option = {#We will keep them
    name = flavor_syr.3.a
    if = {
      limit = {
        SRV = {
          is_subject_of = ROOT
        }
        ARD = {
          is_subject_of = ROOT
        }
      }
      ARD = {
        declare_war_with_cb = {
          who = ROOT
          casus_belli = cb_independence_war
          }
        }
        SRV = {
          create_alliance = ARD
     	 	  join_all_offensive_wars_of = ARD
     	 	}
    }

    if = {
      limit = {
        ARD = {
          is_subject_of = ROOT
        }
        NOT = {
          SRV = {
            is_subject_of = ROOT
          }
        }
      }
      ARD = {
        declare_war_with_cb = {
          who = ROOT
          casus_belli = cb_independence_war
          }
        }
    }
    if = {
      limit = {
        SRV = {
          is_subject_of = ROOT
        }
        NOT = {
          ARD = {
            is_subject_of = ROOT
          }
        }
      }
      SRV = {
        declare_war_with_cb = {
          who = ROOT
          casus_belli = cb_independence_war
          }
        }
    }
  }

  option = {#they will have more freedom
    name = flavor_syr.3.b
    if = {
      limit = {
        SRV = {
          is_subject_of = ROOT
        }
      }
      SRV = {
        create_march = SRV
        add_country_modifier = {
          name = less_betrayed_vassals
          duration = 7300
        }
      }
    }

    if = {
      limit = {
        ARD = {
          is_subject_of = ROOT
        }
      }
      create_march = ARD
      ARD = {
        add_country_modifier = {
          name = less_betrayed_vassals
          duration = 7300
        }
      }
    }
  }
  option = {#they will have to pay for independance
    name = flavor_syr.3.c

      limit = {
        ARD = {
          is_subject_of = ROOT
        }
      }

        grant_independence = yes
      }
    }

    if = {
      limit = {
        SRV = {
          is_subject_of = ROOT
        }
      }
      SRV = {
        grant_independence = yes
      }
    }
  }
}
