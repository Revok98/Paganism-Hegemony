#Cultural change assimilation
country_event = {
  id = flavor_nrt.2
  title = flavor_nrt.2.t
  desc = flavor_nrt.2.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      tag = NRT
      tag = BET
    }
    has_country_modifier = nrt_assimilation
    any_owned_province = {
      religion = norse_pagan_reformed
      OR = {
        culture_group = gaelic
        culture = angle
        culture = seaxan
        culture = britannian
        culture_group = britannian_group
      }
      NOT = { culture_group = scandinavian }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 7
  }

  option = {
    name = flavor_nrt.3.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
        religion = norse_pagan_reformed
        OR = {
          culture_group = gaelic
          culture = angle
          culture = seaxan
          culture = britannian
          culture_group = britannian_group
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      random_list = {
        35 = {
          change_culture = northymbrian
          random_country = {
            limit = {
              any_owned_province = {
                culture = norwegian
                NOT = { has_province_modifier = nrt_migration_affect }
              }
            }
            random_owned_province = {
              limit = {
                culture = norwegian
                NOT = { has_province_modifier = nrt_migration_affect }
              }
              add_province_modifier = {
                name = nrt_migration_affect
                duration = 7300
              }
            }
            add_opinion = {
              who = ROOT
              modifier = opinion_draining_people_nrt
            }
          }
        }
        35 = {
          change_culture = northymbrian
          random_country = {
            limit = {
