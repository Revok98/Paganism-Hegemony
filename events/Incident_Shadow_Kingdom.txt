# Italian country leaves or stays
country_event = {
	id = shadow_kingdom.4
	title = hre_event.1.t
	desc = {
		trigger = {
			has_country_flag = reined_in_for_shadow_kingdom
		}
		desc = shadow_kingdom.4.d2
	}
	desc = {
		trigger = {
			NOT = { has_country_flag = reined_in_for_shadow_kingdom }
		}
		desc = hre_event.1.d
	}
	picture = HRE_eventPicture

	is_triggered_only = yes

	option = { # remain
		ai_chance = {
			factor = 1
			modifier = {
				factor = 1000
				will_leave_for_shadow_kingdom_trigger = no
			}
		}
		name = hre_event.1.b
		if = {
			limit = {
				has_global_flag = emperor_trying_to_avert_shadow_kingdom
			}
			emperor = {
				add_imperial_influence = 5
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = reined_in_for_shadow_kingdom
				}
			}
			add_prestige = -20
			custom_tooltip = shadow_kingdom_triggered_modifier_tt
		}
	}
	option = { # leave
		ai_chance = {
			factor = 1
			modifier = {
				factor = 1000
				will_leave_for_shadow_kingdom_trigger = yes
			}
		}
		trigger = {
			NOT = {
				has_country_flag = reined_in_for_shadow_kingdom
			}
		}
		name = hre_event.1.a
		set_in_empire = no
		set_country_flag = permanently_left_hre_flag
		emperor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_left_hre
			}
		}
		if = {
			limit = {
				any_country = {
					is_subject_of = ROOT
					capital_scope = {
						region = egypt_region
					}
					is_part_of_hre = yes
				}
			}
			every_subject_country = {
				limit = {
					capital_scope = {
						region = egypt_region
					}
					is_part_of_hre = yes
				}
				set_in_empire = no
			}
		}
		if = {
			limit = {
				has_global_flag = emperor_trying_to_avert_shadow_kingdom
			}
			emperor = {
				add_imperial_influence = -10
			}
		}
	}
}
