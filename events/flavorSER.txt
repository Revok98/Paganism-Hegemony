namespace = flavor_ser
country_event = {#bulgar
  id = flavor_ser.1
  title = flavor_ser.1.t
  desc = flavor_ser.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    has_country_flag = cultural_conversion_bulgar_flag
    OR = {
      culture_group = south_slavic
      AND = {
        has_global_flag = panslavism
        culture_group = east_slavic
      }
    }
    any_owned_province = {
      OR = {
        has_owner_religion = yes
        religion = slavic
      }
      OR = {
        culture = bulgar_nomadic
        culture = Thracian
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {
    name = flavor_ser.1.a
    ai_chance = { factor = 1 }
    trigger = {
      NOT = {has_global_flag = panslavism}
    }
    random_owned_province = {
      limit = {
        OR = {
          has_owner_religion = yes
          religion = slavic
        }
        OR = {
          culture = bulgar_nomadic
          culture = Thracian
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = bulgarian
    }
  }
  option = {
    name = flavor_ser.1.a
    ai_chance = { factor = 1 }
    trigger = {
      has_global_flag = panslavism
    }
    random_owned_province = {
      limit = {
        OR = {
          has_owner_religion = yes
          religion = slavic
        }
        OR = {
          culture = bulgar_nomadic
          culture = Thracian
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = bulgarian_united
    }
  }
}
