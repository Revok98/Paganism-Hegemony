		icon = mission_establish_high_seas_navy
		required_missions = { }

		completed_by = 1492.1.1

		trigger = {
			if = {
				limit = {
					is_random_new_world = yes

					tooltip = miss_discover_caribbean

						has_discovered = ROOT

					490 = {
						has_discovered = ROOT
					}
				}
			}
		}

		effect = {
			add_country_modifier = {
				name = "spa_hispaniola_discovered"

			}

	}

					value = 1

			}

		effect = {
			add_prestige = 15

				name = "the_unknown_frontier"

			}
